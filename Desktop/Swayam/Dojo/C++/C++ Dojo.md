using namespace std;
int gcd(int a, int b) {
	while (b != 0) {
		int temp = b;
        b = a % b;

}  
long long lcm(int a, int b) {
	if (a == 0 || b == 0) return 0;   // handle zero case     
		return (1LL * a * b) / gcd(a, b); // use long long to avoid overflow }
